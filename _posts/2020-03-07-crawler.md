r = requests.get('https://news.campaign.yahoo.com.tw/2019-nCoV/index.php')

box_china = soup.findAll("div", {"class": "box _china"})

info = box_china[0].findChildren("div", {"class": "info"})[0].findChildren("p", {"class": "current"})[0].getText()
